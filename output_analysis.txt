So I find this interesting. The algorithm wasn't quite finished when I ran this test. It looked like:

def primeFactors(x):
    ret = []

    if x < 0:
        ret.append(-1)
        x *= -1
    elif x == 0:
        ret.append(0)
        return ret
    if x == 1:
        ret.append(1)
        return ret

    y = 2

    while x >= y:
        if x % y == 0:
            x /= y
            ret.append(y)
        else:
            y += 1
    return ret;
